жҺҘдёӢжқҘз»ҷеӨ§е®¶иҙҙдёҖдёӢзӨәдҫӢжәҗз Ғ,иҝҷдёӘжәҗз ҒжқҘжәҗдәҺDubboжәҗз Ғзӣ®еҪ•зҡ„ dubbo-demo/dubbo-demo-api зӣ®еҪ•дёӢйқўзҡ„dubbo-demo-api-providerеӯҗйЎ№зӣ®,иҝҷйҮҢжҲ‘еҒҡдәҶеҲ еҮҸ,ж–№дҫҝзңӢж ёеҝғд»Јз Ғ:
йҰ–е…ҲжҲ‘д»¬е®ҡд№үдёҖдёӘжңҚеҠЎжҺҘеҸЈеҰӮдёӢжүҖзӨә:

import java.util.concurrent.CompletableFuture;
public interface DemoService {
    /**
     * еҗҢжӯҘеӨ„зҗҶзҡ„жңҚеҠЎж–№жі•
     * @param name
     * @return
     */
    String sayHello(String name);

    /**
     * з”ЁдәҺејӮжӯҘеӨ„зҗҶзҡ„жңҚеҠЎж–№жі•
     * @param name
     * @return
     */
    default CompletableFuture<String> sayHelloAsync(String name) {
        return CompletableFuture.completedFuture(sayHello(name));
    }
}

1.3 еҗҜз”ЁжңҚеҠЎ

жңүдәҶжңҚеҠЎжҺҘеҸЈд№ӢеҗҺжҲ‘д»¬жқҘеҗҜз”ЁжңҚеҠЎ,еҗҜз”ЁжңҚеҠЎзҡ„жәҗз ҒеҰӮдёӢ:

import org.apache.dubbo.common.constants.CommonConstants;
import org.apache.dubbo.config.ApplicationConfig;
import org.apache.dubbo.config.MetadataReportConfig;
import org.apache.dubbo.config.ProtocolConfig;
import org.apache.dubbo.config.RegistryConfig;
import org.apache.dubbo.config.ServiceConfig;
import org.apache.dubbo.config.bootstrap.DubboBootstrap;
import org.apache.dubbo.demo.DemoService;

public class Application {
    public static void main(String[] args) throws Exception {
            startWithBootstrap();
    }
    private static void startWithBootstrap() {
        ServiceConfig<DemoServiceImpl> service = new ServiceConfig<>();
        service.setInterface(DemoService.class);
        service.setRef(new DemoServiceImpl());
        DubboBootstrap bootstrap = DubboBootstrap.getInstance();
        bootstrap.application(new ApplicationConfig("dubbo-demo-api-provider"))
            .registry(new RegistryConfig("zookeeper://127.0.0.1:2181"))
            .protocol(new ProtocolConfig(CommonConstants.DUBBO, -1))
            .service(service)
            .start()
            .await();
    }
}

1.4 еҗҜз”ЁжңҚеҠЎеҗҺеҶҷе…ҘZookeeperзҡ„иҠӮзӮ№ж•°жҚ®

еҗҜеҠЁжңҚеҠЎ,иҝҷдёӘж—¶еҖҷжҲ‘д»¬жү“ејҖZookeeperеӣҫеҪўеҢ–е®ўжҲ·з«ҜжқҘзңӢзңӢиҝҷдёӘжңҚеҠЎеңЁZookeeperдёҠйқўеҶҷе…ҘжқҘе“Әдәӣж•°жҚ®,еҰӮдёӢеӣҫ:

еҶҷе…ҘZookeperдёҠзҡ„иҠӮзӮ№з”ЁдәҺжңҚеҠЎеңЁеҲҶеёғејҸеңәжҷҜдёӢзҡ„еҚҸи°ғ,иҝҷдәӣиҠӮзӮ№жҳҜжҜ”иҫғйҮҚиҰҒзҡ„гҖӮ

еҰӮжһңдәҶи§ЈиҝҮDubboзҡ„еҗҢеӯҰ,еә”иҜҘдјҡзҹҘйҒ“DubboеңЁдҪҺзүҲжң¬зҡ„ж—¶еҖҷдјҡеҗ‘жіЁеҶҢдёӯеҝғдёӯеҶҷе…ҘжңҚеҠЎжҺҘеҸЈ,е…·дҪ“и·Ҝеҫ„еңЁдёҠйқўзҡ„ dubboзӣ®еҪ•дёӢ ,з„¶еҗҺеңЁ /dubbo/жңҚеҠЎжҺҘеҸЈ/ и·Ҝеҫ„дёӢеҶҷе…ҘеҰӮдёӢдҝЎжҒҜ:

  • жңҚеҠЎжҸҗдҫӣиҖ…й…ҚзҪ®дҝЎжҒҜURLеҪўејҸ
  • жңҚеҠЎж¶Ҳиҙ№иҖ…зҡ„й…ҚзҪ®дҝЎжҒҜURLеҪўејҸ
  • жңҚеҠЎи·Ҝз”ұдҝЎжҒҜ

дёҠйқўиҝҷдёӘеӣҫе°ұжҳҜDubbo3зҡ„жіЁеҶҢдҝЎжҒҜдәҶ,еҗҺйқўжҲ‘д»¬д№ҹдјҡеӣҙз»•з»ҶиҠӮжқҘиҜҙжҳҺдёӢ,иҝҷйҮҢеҸҜд»ҘзңӢдёӢж–°еўһдәҶ:
